# Para archivos específicos que deben usar LF
*.sh text eol=lf
*.java text eol=lf
*.properties text eol=lf
*.xml text eol=lf

# Para archivos específicos que deben usar CRLF (generalmente scripts de Windows)
*.cmd text eol=crlf
*.bat text eol=crlf
